import tkinter as tk
from tkinter import messagebox

# 1. OOP: كلاس المريض (الأساس)
class Hasta:
    def __init__(self, ad, telefon, sikayet):
        self.ad = ad
        self.telefon = telefon
        self.sikayet = sikayet

# 2. واجهة البرنامج (بالتركي حسب طلب الجامعة)
class KlinikApp:
    def __init__(self, pencere):
        self.pencere = pencere
        self.pencere.title("Diş Kliniği Yönetim Sistemi")
        self.pencere.geometry("400x300")

        # العنوان
        self.label = tk.Label(pencere, text="Hasta Kayıt Sistemi", font=("Arial", 16))
        self.label.pack(pady=20)

        # زر الإضافة
        self.kayit_btn = tk.Button(pencere, text="Yeni Hasta Ekle", command=self.ekle_mesaj)
        self.kayit_btn.pack(pady=10)

    def ekle_mesaj(self):
        messagebox.showinfo("Bilgi", "Hasta başarıyla kaydedildi!")

# تشغيل البرنامج
if __name__ == "__main__":
    root = tk.Tk()
    app = KlinikApp(root)
    root.mainloop()
